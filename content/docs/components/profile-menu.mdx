## Usage

```tsx
import {
  ProfileMenu,
  ProfileMenuContent,
  ProfileMenuGroup,
  ProfileMenuHeader,
  ProfileMenuHeaderContent,
  ProfileMenuItem,
  ProfileMenuTrigger,
} from "@/components/molecule-ui/profile-menu"
```

```tsx
<ProfileMenu>
  <ProfileMenuHeader>
    <ProfileMenuHeaderContent />
    <ProfileMenuTrigger />
  <ProfileMenuHeader>
  <ProfileMenuContent>
    <ProfileMenuGroup>
      <ProfileMenuItem />
    </ProfileMenuGroup>
  </ProfileMenuContent>
</ProfileMenuHeader>
```
