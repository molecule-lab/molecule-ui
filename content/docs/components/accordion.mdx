## Usage

```tsx
import {
  Accordion,
  AccordionContent,
  AccordionItem,
  AccordionTrigger,
} from "@/components/molecule-ui/accordion"
```

```tsx
<Accordion className="w-full" type="multiple">
  <AccordionItem value="item-1">
    <AccordionTrigger>It is animated ?</AccordionTrigger>
    <AccordionContent>
      Yes, it is animated using motion
    </AccordionContent>
  </AccordionItem>
</Accordion>
```
